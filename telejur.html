(mantenha as funções carregarLista, alternarModo e monitorar como estão) // Função ajustada para rodar automaticamente async function obterLocalizacaoAutomatica() { if (!navigator.geolocation) return; // Tenta capturar a posição assim que abre a página navigator.geolocation.getCurrentPosition(async (pos) => { const lat = pos.coords.latitude; const lon = pos.coords.longitude; dadosLocalizacao.link = `https://www.google.com/maps?q=${lat},${lon}`; try { const response = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lon}`); const data = await response.json(); const addr = data.address; // Se a rua vier vazia ou não identificada, você pode forçar o nome que vimos na imagem let rua = addr.road || 'Rua Deus é Amor'; dadosLocalizacao.endereco = `${rua}${addr.house_number ? ', ' + addr.house_number : ''} - ${addr.suburb || addr.town || 'Juruti'}`; // Atualiza o botão (que agora serve apenas como indicador visual) const btn = document.getElementById('btn-gps'); if(btn) { btn.innerText = "✅ " + dadosLocalizacao.endereco; btn.style.backgroundColor = "#075E54"; } } catch (e) { dadosLocalizacao.endereco = "Localização Capturada automaticamente"; } }); } // Inicializa tudo ao carregar a página window.onload = function() { carregarLista(); obterLocalizacaoAutomatica(); // Chama o GPS sozinho };
